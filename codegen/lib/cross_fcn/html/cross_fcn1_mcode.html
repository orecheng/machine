1function [ AveAirMoistureOut,AveAirTempratureOut,AveSolTempratureOut,AveSolMassOut,AveSolConcenOut,AveSolEnthalpyOut] = cross_fcn( Ta_in,phi,Ts_in,Ps_in,Ma_in,Ms_in)
  2% Ta_in=30;%
  3% phi=0.8;%
  4
  5T=Ta_in+273.15;
  6[rho_air,da_in,ha_in]= rh2da(Ta_in,phi);
  7% Ts_in=10;%
  8% Ps_in=0.3;%
  9rho_licl=cal_rho_licl(Ts_in,Ps_in);
 10
 11% Va_in=4000;%
 12% Vs_in=4.8;%
 13
 14% Ms_in=Vs_in;
 15% Ma_in=Va_in;
 16
 17H=0.5;%
 18L=0.15;%
 19W=0.65;%
 20
 21hs_in=sol_enthalpy(Ts_in,Ps_in);
 22
 23meshgrid=0.005;
 24M=ceil(L/meshgrid);
 25N=ceil(H/meshgrid);
 26
 27V=H*L*W;
 28Fa=Ma_in/(H*W);
 29Fz=Ms_in/(W*L);
 30
 31%% 水在空气中的扩散系数
 32Dair=(-0.29890+1.6253e-3*T+7.5e-7*T^2)*1e-4;
 33%% 空气的动力粘度和运动粘度
 34mu_da=[-2.448e-6,0.005072,1.713]*[Ta_in^2;Ta_in;1]*1e-5; %动力粘度
 35nv_da=mu_da/rho_air; %运动粘度
 36%% Thermal Conductivity of Air
 37lamda=[6.993e-8,0.007618,2.442]*[Ta_in^2;Ta_in;1]*1e-2;
 38%% Sc
 39Sc=nv_da/Dair;
 40%% Pr
 41Cpa=1.01e3;
 42Pr=mu_da*Cpa/lamda;
 43Le=Sc/Pr;
 44%% NTU
 45%% 我们自己实验数据拟合的
 46% if Ts_in>35
 47%     a=[0.010296438	-0.833147603	1.180504199	-23.45078711];%再生
 48% else
 49%     a=[5.094975587	-0.061006773	0.609178325	-1.330942842];%除湿
 50% end
 51%% 清华大学刘拴强博士毕业论文拟合的
 52if Ts_in>35
 53    a=[0.075344 0.9697 0.619 -5.3885];%再生
 54else
 55    a=[5.088 0.7329	0.4203 1.5328];%除湿
 56end
 57
 58NTU=a(1)*V*Fa^a(2)*Fz^a(3)*(1-Ps_in)^a(4);%
 59%% Initial of Matrix
 60SolEnthalpy=zeros(N,M);
 61AirMoisture=zeros(N,M);
 62AirEnthalpy=zeros(N,M);
 63AirTemprature=zeros(N,M);
 64SolTemprature=zeros(N,M);
 65SolMass=zeros(N,M);
 66SolConcen=zeros(N,M);
 67%% row_1 column_1
 68[AirMoisture(1,1),AirTemprature(1,1),AirEnthalpy(1,1),SolTemprature(1,1),SolMass(1,1),SolConcen(1,1),SolEnthalpy(1,1)]= cross_core(Ts_in,Ps_in,Ms_in/M,ha_in,da_in,Ma_in/N,M,N,NTU,Le);
 69%% row_1
 70for j=2:M
 71    [AirMoisture(1,j),AirTemprature(1,j),AirEnthalpy(1,j),SolTemprature(1,j),SolMass(1,j),SolConcen(1,j),SolEnthalpy(1,j)]= cross_core(Ts_in,Ps_in,Ms_in/M,AirEnthalpy(1,j-1),AirMoisture(1,j-1),Ma_in/N,M,N,NTU,Le);
 72end
 73%% column_1
 74for i=2:N
 75    [AirMoisture(i,1),AirTemprature(i,1),AirEnthalpy(i,1),SolTemprature(i,1),SolMass(i,1),SolConcen(i,1),SolEnthalpy(i,1)]= cross_core(SolTemprature(i-1,1),SolConcen(i-1,1),SolMass(i-1,1),ha_in,da_in,Ma_in/N,M,N,NTU,Le);
 76end
 77%% row_2~row_end  column_1~column_end
 78for i=2:N
 79    for j=2:M
 80        [AirMoisture(i,j),AirTemprature(i,j),AirEnthalpy(i,j),SolTemprature(i,j),SolMass(i,j),SolConcen(i,j),SolEnthalpy(i,j)]= cross_core(SolTemprature(i-1,j),SolConcen(i-1,j),SolMass(i-1,j),AirEnthalpy(i,j-1),AirMoisture(i,j-1),Ma_in/N,M,N,NTU,Le);
 81    end
 82end
 83
 84%% Average Out
 85AveAirMoistureOut =mean(AirMoisture(1:N,M));
 86AveAirEnthalpyOut =mean(AirEnthalpy(1:N,M));
 87AveAirTempratureOut =mean(AirTemprature(1:N,M));
 88
 89AveSolTempratureOut =mean(SolTemprature(N,1:M));
 90AveSolMassOut=sum(SolMass(N,1:M));
 91AveSolConcenOut=mean(SolConcen(N,1:M));
 92AveSolEnthalpyOut=mean(SolEnthalpy(N,1:M));
 93
 94err  = (ha_in-AveAirEnthalpyOut)*Ma_in/(sum(SolEnthalpy(N,1:M).*SolMass(N,1:M))-hs_in*Ms_in)-1;
 95err2 = (da_in-AveAirMoistureOut)*Ma_in/(sum(SolMass(N,1:M))-Ms_in)-1;
 96
 97
 98end

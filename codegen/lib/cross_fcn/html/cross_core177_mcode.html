1function [da_out,Ta_out,ha_out,Ts_out,Ms_out,Ps_out,hs_out]= cross_core(Ts_in,Ps_in,Ms_in,ha_in,da_in,Ma_in,M,N,NTU,Le)
 2
 3A=2-(1+(Ps_in/0.28)^4.3)^0.6;
 4B=(1+(Ps_in/0.21)^5.1)^0.49-1;
 5Fpt=A+B*(((Ts_in+273.15)/228)-1);
 6Pi25=1-(1+(Ps_in/0.362)^(-4.75))^(-0.4)-0.03*exp(-(((Ps_in-0.1)^2)/0.005));
 7t=1-(Ts_in+273.15)/647.14;
 8E=exp(((-7.85823)*t+1.83991*(t^1.5)+(-11.7811)*(t^3)+22.6705*(t^3.5)+(-15.9393)*(t^4)+1.77516*(t^7.5))/(1-t));
 9Pw=2.212*10^7*E;
10Psol=Pw*Pi25*Fpt;                                %溶液表面的蒸汽压
11de_in=0.622*(Psol/(101325-Psol));    %与溶液状态相平衡的饱和空气含湿量
12%% 计算溶液焓值
13X=Ps_in*100;
14a=[-66.2324,11.2711,-0.79853,2.1534e-2,-1.66352e-4];
15b=[4.5751,-0.146924,6.307226e-3,-1.38054e-4,1.06690e-6];
16c=[-8.09689e-4,2.18145e-4,-1.36194e-5,3.20998e-7,-2.64266e-9];
17x=[1;X;X^2;X^3;X^4];
18A=a*x;
19B=b*x;
20C=c*x;
21hs_in=A+B*Ts_in+C*Ts_in^2;
22%% he,i-1 换算 Tsin de,i-1 
23he_in=1.01*Ts_in+(2501+1.83*Ts_in)*de_in;
24%% da,i ①
25da_out=da_in+NTU*(de_in-da_in)/M;
26%% ha,i ②
27rts_in=2500-2.35*Ts_in;
28ha_out=ha_in+NTU*Le*(he_in-ha_in+rts_in*(1/Le-1)*(de_in-da_in))/M;
29Ta_out=(ha_out -2501*da_out)/(1.01+1.83*da_out);
30%% Ms,j ③
31Ms_out=Ms_in+Ma_in*(da_in-da_out)*M/N;
32%% Ps,j ④
33Ps_out=Ps_in*Ms_in/Ms_out;
34%% hs,j ⑤
35hs_out=(hs_in*Ms_in-Ma_in*(ha_out-ha_in)*M/N)/Ms_out;
36%% Ts,j
37X=Ps_out*100;
38a=[-66.2324,11.2711,-0.79853,2.1534e-2,-1.66352e-4];
39b=[4.5751,-0.146924,6.307226e-3,-1.38054e-4,1.06690e-6];
40c=[-8.09689e-4,2.18145e-4,-1.36194e-5,3.20998e-7,-2.64266e-9];
41x=[1;X;X^2;X^3;X^4];
42A=a*x;
43B=b*x;
44C=c*x;
45Ts_out=(-B+sqrt(B^2-4*C*(A-hs_out)))/(2*C);
46end

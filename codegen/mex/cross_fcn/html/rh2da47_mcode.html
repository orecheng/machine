1function [rho_air,da,ha]= rh2da(t,phi)
 2% t=41.1;
 3% phi=0.649;
 4if phi>10
 5phi=phi/100;
 6end
 7T=t+273.15;
 8Pma=101325;
 9Rda=287;
10ps=exp(7.23e-7*t^3-2.71e-4*t^2+7.2e-2*t+6.42);
11da=0.622*phi*ps/(Pma-phi*ps);
12rho_air=Pma*(1+0.001*da)/(Rda*T*(1+0.001606*da));% 湿空气密度 （kg/kg干空气）
13ha=1.01*t+(1.83*t+2501)*da;
14end
